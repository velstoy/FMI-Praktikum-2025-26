int n;
std::cin >> n;

int** matrix = new (std::nothrow) int* [n];

if (!matrix)
{
	return -1;
}

for (int i = 0; i < n; i++)
{
	matrix[i] = new (std::nothrow) int[n];

	if (!matrix[i])
	{
		return -1;
	}

	for (int j = 0; j < n; j++)
	{
		std::cin >> matrix[i][j];
	}
}

std::cout << sumDigitsAboveMainDiagonal(matrix, n) << std::endl;

for (int i = 0; i < n; i++)
{
	delete[] matrix[i];
	matrix[i] = nullptr;
}

delete[] matrix;
matrix = nullptr;
