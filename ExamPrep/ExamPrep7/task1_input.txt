size_t size;
std::cin >> size;

unsigned int* arr = new (std::nothrow) unsigned int[size];

if (!arr)
{
	std::cout << "Error allocating memory for array!" << std::endl;
	return -1;
}

for (size_t i = 0; i < size; i++)
{
	std::cin >> arr[i];
}

unsigned int gcd = getGCD(arr, size);

if (gcd == -1)
{
	std::cout << "Could not find gcd of array elements!" << std::endl;
	return -1;
}

std::cout << gcd << std::endl;

delete[] arr;
